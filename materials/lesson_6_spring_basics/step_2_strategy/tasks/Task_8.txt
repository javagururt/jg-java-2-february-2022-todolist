Объявите в классе ProgramMenu конструктор следующим образом:


public class ProgramMenu {

	private Map<Integer, UIAction> menuNumberToUIActionMap;

	@Autowired
	public ProgramMenu(List<UIAction> uiActions) {

	}

	public void print() {
		System.out.println();
		System.out.println("Program menu:");
		System.out.println("1. Add book to list");
		System.out.println("2. Delete book from list");
		System.out.println("3. Show all books in the list");
		System.out.println("4. Search books");
		System.out.println("5. Exit");
		System.out.println();
	}

	public int getMenuNumberFromUser() {
		System.out.println("Enter menu item number to execute:");
		Scanner scanner = new Scanner(System.in);
		return Integer.parseInt(scanner.nextLine());
	}

}

С помощью такого конструктора можно получить все инстарнции классов
которые реализуют интерфейс UIAction и являются компонентами
(на которых проставлена аннотация @Component).
